Kapitel 4 – Abstraktion und Material
Silke Miche („Kästen“, 2010) erkundet das Zusammenspiel von Form, Farbe und Fläche, während Gerd Sonntags „Zunge, orange“ (2011) Glas, Messing und Kristall in ein poetisch-humorvolles Objekt verwandelt.

import pandas as pd

def simple_backtest(prices: pd.DataFrame, signal: pd.Series, fee_bps=1.0):
    # align
    df = prices[["date","symbol","close"]].copy()
    df["signal"] = signal.reindex(df.index)
    # very naive: next-day open/close returns with signal as weight
    df["ret"] = df.groupby("symbol")["close"].pct_change().shift(-1)
    df["gross"] = df["signal"] * df["ret"]
    df["fees"] = (abs(df["signal"].diff().fillna(0)) * fee_bps/1e4)
    df["net"] = df["gross"] - df["fees"]
    return df

from pathlib import Path
from datetime import datetime
import argparse, re

ROOT = Path(__file__).resolve().parents[1]
CTX  = ROOT / "BerlinAndBeyond" / "context.md"

def replace_token(text, token, value):
    return re.sub(r"\{\{"+re.escape(token)+r"\}\}", value, text)

def stamp_now_utc():
    return datetime.utcnow().strftime("%Y-%m-%d %H:%M UTC")

def main():
    p = argparse.ArgumentParser(description="Update Berlin & Beyond context.md")
    p.add_argument("--batches-done", default=None, help='e.g. "01–03 (works 1–30)"')
    p.add_argument("--current-batch", default=None, help='e.g. "04 (works 31–40)"')
    p.add_argument("--next-step", default=None, help='Plain sentence, what we do next')
    args = p.parse_args()

    text = CTX.read_text(encoding="utf-8")

    text = replace_token(text, "STAMP", stamp_now_utc())
    if args.batches_done:
        text = replace_token(text, "BATCHES_DONE", args.batches_done)
    if args.current_batch:
        text = replace_token(text, "CURRENT_BATCH", args.current_batch)
    if args.next_step:
        text = replace_token(text, "NEXT_STEP", args.next_step)

    CTX.write_text(text, encoding="utf-8")
    print(f"Updated {CTX}")

if __name__ == "__main__":
    main()

2) Streamlit App (control_panel.py)
import os
import re
import subprocess
from datetime import datetime, timezone
from pathlib import Path

import streamlit as st
import yaml
import requests

# ---------- Config ----------
REPO_ROOT = Path(__file__).resolve().parent
CONFIG_PATH = REPO_ROOT / "projects_config.yaml"

def load_config():
    with open(CONFIG_PATH, "r", encoding="utf-8") as f:
        return yaml.safe_load(f)["projects"]

def read_text(path: Path) -> str:
    return path.read_text(encoding="utf-8") if path.exists() else ""

def write_text(path: Path, text: str):
    path.parent.mkdir(parents=True, exist_ok=True)
    path.write_text(text, encoding="utf-8")

def git_run(args):
    """Run a git command and return (ok, out)."""
    try:
        res = subprocess.run(["git"] + args, cwd=REPO_ROOT, capture_output=True, text=True, check=False)
        ok = (res.returncode == 0)
        out = res.stdout + "\n" + res.stderr
        return ok, out.strip()
    except Exception as e:
        return False, str(e)

def replace_token(text: str, token: str, value: str) -> str:
    # replace {{TOKEN}} with value; if not found, try to insert a sensible default line
    if "{{" + token + "}}" in text:
        return text.replace("{{" + token + "}}", value)
    return text

def stamp_now_utc() -> str:
    return datetime.now(timezone.utc).strftime("%Y-%m-%d %H:%M UTC")

def extract_field(text: str, label: str) -> str:
    """
    Try to extract a value line like:
    **Batches completed:** 01–03 (works 1–30)
    """
    pattern = rf"\*\*{re.escape(label)}\*\*:\s*(.*)"
    m = re.search(pattern, text)
    return m.group(1).strip() if m else ""

def update_field_block(text: str, label: str, value: str) -> str:
    pattern = rf"(\*\*{re.escape(label)}\*\*:\s*)(.*)"
    repl = rf"\1{value}"
    if re.search(pattern, text):
        return re.sub(pattern, repl, text)
    # if not found, append under header
    insertion = f"**{label}**: {value}\n"
    return text.strip() + "\n" + insertion

# ---------- UI ----------
st.set_page_config(page_title="Project Control Panel", layout="wide")
st.title("🧭 Project Control Panel")

projects = load_config()
project_names = list(projects.keys())
selected = st.sidebar.selectbox("Select project", project_names, index=project_names.index("BerlinAndBeyond") if "BerlinAndBeyond" in project_names else 0)
cfg = projects[selected]

st.sidebar.markdown("### Quick Links")
if cfg.get("master_view"):
    st.sidebar.markdown(f"- **Master (view):** [{cfg['master_view']}]({cfg['master_view']})")
if cfg.get("master_raw"):
    st.sidebar.markdown(f"- **Master (raw):** [{cfg['master_raw']}]({cfg['master_raw']})")
if cfg.get("images_view"):
    st.sidebar.markdown(f"- **Images:** [{cfg['images_view']}]({cfg['images_view']})")

context_path = REPO_ROOT / cfg["context_path"]
context_text = read_text(context_path)

st.subheader(f"{selected} — context.md")
if not context_text:
    st.info("No context.md found yet. Paste your template below and click **Create / Update**.")

# Pre-fill fields by trying to read from context.md
stamp_current = stamp_now_utc()
batches_done = extract_field(context_text, "Batches completed")
current_batch = extract_field(context_text, "Current batch")
next_step = extract_field(context_text, "Next Step")

col1, col2, col3 = st.columns(3)
with col1:
    batches_done = st.text_input("Batches completed", value=batches_done or "01–03 (works 1–30)")
with col2:
    current_batch = st.text_input("Current batch", value=current_batch or "04 (works 31–40)")
with col3:
    next_step = st.text_input("Next step", value=next_step or "Describe works 31–40 in final format.")

with st.expander("🔧 Edit full context.md (optional)"):
    context_text = st.text_area("Context content", value=context_text, height=300)

colA, colB, colC = st.columns([1,1,2])
with colA:
    do_git = st.checkbox("git add + commit + push", value=True)
with colB:
    commit_msg = st.text_input("Commit message", value=f"context({selected}): update status {stamp_current}")

def apply_updates(raw_text: str) -> str:
    text = raw_text or ""
    # Ensure key fields exist/updated
    text = update_field_block(text, "Date/Time (UTC)", stamp_current)
    text = update_field_block(text, "Batches completed", batches_done)
    text = update_field_block(text, "Current batch", current_batch)
    text = update_field_block(text, "Next Step", next_step)
    return text

if st.button("✅ Create / Update context.md"):
    new_text = apply_updates(context_text)
    write_text(context_path, new_text)
    st.success(f"Updated {context_path.relative_to(REPO_ROOT)}")

    if do_git:
        ok1, out1 = git_run(["add", str(context_path.relative_to(REPO_ROOT))])
        ok2, out2 = git_run(["commit", "-m", commit_msg])
        ok3, out3 = git_run(["push"])
        st.code(out1 + "\n" + out2 + "\n" + out3)
        if ok3:
            st.success("Pushed to remote.")
        else:
            st.warning("Push may have failed—check output above.")

# Optional: preview master.md (read-only)
st.subheader("📄 Master.md preview (read-only)")
raw = cfg.get("master_raw", "")
if raw:
    try:
        r = requests.get(raw, timeout=10)
        if r.ok:
            st.code(r.text[:8000])  # show first ~8k chars
        else:
            st.warning(f"Could not fetch master_raw ({r.status_code}).")
    except Exception as e:
        st.warning(f"Preview error: {e}")
else:
    st.info("No master_raw URL configured yet for this project.")
